import React from 'react';
import { ChevronUp, ChevronDown } from 'lucide-react';

export default function ScrollablePicker({ value, onChange, max, label }) {
  const items = Array.from({ length: max + 1 }, (_, i) => i);
  const scrollRef = React.useRef(null);

  const handleIncrement = () => {
    const newValue = value < max ? value + 1 : 0;
    onChange(newValue);
  };

  const handleDecrement = () => {
    const newValue = value > 0 ? value - 1 : max;
    onChange(newValue);
  };

  return (
    <div className="flex flex-col items-center">
      <span className="text-xs font-semibold text-gray-600 mb-2">{label}</span>
      <div className="relative">
        <button
          type="button"
          onClick={handleIncrement}
          className="absolute -top-8 left-1/2 -translate-x-1/2 p-1 hover:bg-gray-100 rounded transition-colors z-10"
        >
          <ChevronUp className="w-5 h-5 text-gray-600" />
        </button>
        
        <div 
          ref={scrollRef}
          className="h-32 w-20 overflow-y-auto hide-scrollbar relative"
          style={{
            scrollSnapType: 'y mandatory',
            scrollBehavior: 'smooth'
          }}
